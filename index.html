🧮 Mifflin-St Jeor

The most accurate modern BMR formula. Recommended for most people.

// Male
BMR = 10W + 6.25H − 5A + 5
// Female
BMR = 10W + 6.25H − 5A − 161

📐 Harris-Benedict

Classic formula revised in 1984. Widely used before Mifflin-St Jeor.

// Male
BMR = 13.4W + 4.8H − 5.68A + 88.4
// Female
BMR = 9.25W + 3.1H − 4.33A + 447.6

🎯 What is TDEE?

Total Daily Energy Expenditure (TDEE) = BMR × Activity Factor. It represents the total calories burned in a day including all activity.

⚖️ Weight Goals

1 lb ≈ 3,500 calories. A deficit of 500 cal/day loses ~1 lb/week. A surplus of 500 cal/day gains ~1 lb/week. Never cut more than 1,000 cal/day.

calorie_calculator.cpp
// CalcFit — Calorie Calculator in C++
// Uses Mifflin-St Jeor & Harris-Benedict Equations

#include <iostream>
#include <string>
#include <iomanip>
using namespace std;

struct UserData {
    double weight_kg;
    double height_cm;
    int    age;
    char   gender;       // 'M' or 'F'
    double activity;     // 1.2 - 1.9
};

// Mifflin-St Jeor Equation
double calcBMR_Mifflin(const UserData& u) {
    double bmr = 10*u.weight_kg + 6.25*u.height_cm - 5*u.age;
    return (u.gender == 'M') ? bmr + 5 : bmr - 161;
}

// Harris-Benedict Equation (Revised)
double calcBMR_Harris(const UserData& u) {
    if (u.gender == 'M')
        return 13.397*u.weight_kg + 4.799*u.height_cm - 5.677*u.age + 88.362;
    return 9.247*u.weight_kg + 3.098*u.height_cm - 4.330*u.age + 447.593;
}

void printResults(double bmr, double activity) {
    double tdee = bmr * activity;
    cout << fixed << setprecision(0);
    cout << "\n===== RESULTS =====\n";
    cout << "BMR       : " << bmr  << " kcal/day\n";
    cout << "Maintain  : " << tdee << " kcal/day\n";
    cout << "Lose 1lb  : " << tdee-500 << " kcal/day\n";
    cout << "Gain 1lb  : " << tdee+500 << " kcal/day\n";
}

int main() {
    UserData user;
    cout << "=== CalcFit: Calorie Calculator ===\n\n";
    cout << "Weight (kg): ";  cin >> user.weight_kg;
    cout << "Height (cm): ";  cin >> user.height_cm;
    cout << "Age:         ";  cin >> user.age;
    cout << "Gender (M/F): "; cin >> user.gender;
    cout << "Activity (1.2-1.9): "; cin >> user.activity;

    double bmr = calcBMR_Mifflin(user);
    printResults(bmr, user.activity);
    return 0;
}
